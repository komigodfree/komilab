{{/* Shortcode: steps Usage: {{< steps >}} 1. **Titre étape** — Description 2. **Titre étape 2** — Description {{< /steps >}} */}}
{{- $raw := trim .Inner "\n\r\t " -}} {{- $lines := split $raw "\n" -}} {{- range $i, $l := $lines -}} {{- $line := trim $l "\n\r\t " -}} {{- if gt (len $line) 0 -}} {{- $num := add $i 1 -}} {{- $content := replaceRE `^\d+\.\s*` "" $line -}} {{- $parts := split $content " — " -}}
{{ $num }}
{{- if gt (len $parts) 1 -}}
{{ index $parts 0 | markdownify }}
{{ index $parts 1 | markdownify }}
{{- else -}}
{{ $content | markdownify }}
{{- end -}}
{{- end -}} {{- end -}}
